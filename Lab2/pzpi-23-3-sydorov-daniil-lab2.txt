package com.example.minishop

import android.content.Intent
import android.os.Bundle
import android.widget.Button
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val btnPhone = findViewById<Button>(R.id.btnPhone)
        val btnTablet = findViewById<Button>(R.id.btnTablet)
        val btnHeadphones = findViewById<Button>(R.id.btnHeadphones)
        val btnCart = findViewById<Button>(R.id.btnCart)

        btnPhone.setOnClickListener {
            CartManager.addItem("Iphone 17 Pro")
            Toast.makeText(this, "Телефон було додано", Toast.LENGTH_SHORT).show()
        }

        btnTablet.setOnClickListener {
            CartManager.addItem("iPad Air")
            Toast.makeText(this, "Планшет було додано", Toast.LENGTH_SHORT).show()
        }

        btnHeadphones.setOnClickListener {
            CartManager.addItem("AirPods Pro")
            Toast.makeText(this, "Навушники були додані", Toast.LENGTH_SHORT).show()
        }

        btnCart.setOnClickListener {
            startActivity(Intent(this, CartActivity::class.java))
        }
    }
}

B2. CartActivity.kt
package com.example.minishop

import android.content.Intent
import android.os.Bundle
import android.widget.ArrayAdapter
import android.widget.Button
import android.widget.ListView
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class CartActivity : AppCompatActivity() {

    private lateinit var adapter: ArrayAdapter<String>

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_cart)

        val listView = findViewById<ListView>(R.id.listViewCart)
        val btnCheckout = findViewById<Button>(R.id.btnCheckout)

        adapter = ArrayAdapter(
            this,
            android.R.layout.simple_list_item_1,
            CartManager.items
        )

        listView.adapter = adapter

        listView.setOnItemLongClickListener { _, _, position, _ ->
            val item = CartManager.items[position]
            CartManager.items.removeAt(position)
            adapter.notifyDataSetChanged()

            Toast.makeText(this, "$item Видален", Toast.LENGTH_SHORT).show()
            true
        }

        btnCheckout.setOnClickListener {
            if (CartManager.items.isEmpty()) {
                Toast.makeText(this, "Кошик пустий", Toast.LENGTH_SHORT).show()
                return@setOnClickListener
            }

            startActivity(Intent(this, CheckoutActivity::class.java))
        }
    }

    override fun onResume() {
        super.onResume()
        adapter.notifyDataSetChanged()
    }
}

B3. CheckoutActivity.kt
package com.example.minishop

import android.content.Intent
import android.os.Bundle
import android.widget.Button
import android.widget.EditText
import androidx.appcompat.app.AppCompatActivity

class CheckoutActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_checkout)

        val etName = findViewById<EditText>(R.id.etName)
        val etPhone = findViewById<EditText>(R.id.etPhone)
        val etCity = findViewById<EditText>(R.id.etCity)
        val btnOrder = findViewById<Button>(R.id.btnOrder)

        btnOrder.setOnClickListener {

            if (etName.text.isEmpty() ||
                etPhone.text.isEmpty() ||
                etCity.text.isEmpty()
            ) return@setOnClickListener

            CartManager.clearCart()

            val intent = Intent(this, ThankYouActivity::class.java)
            intent.flags = Intent.FLAG_ACTIVITY_NEW_TASK or Intent.FLAG_ACTIVITY_CLEAR_TASK
            startActivity(intent)
        }
    }
}

package com.example.minishop

object CartManager {
    val items = mutableListOf<String>()

    fun addItem(item: String) {
        items.add(item)
    }

    fun clearCart() {
        items.clear()
    }
}
